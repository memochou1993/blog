## 做法

新增 `main.go` 檔。

```GO
package main

import (
	"bufio"
	"log"
	"net"
)

func main() {
	// 建立監聽器
	ln, err := net.Listen("tcp", ":8080")
	if err != nil {
		log.Fatalln(err)
	}
	// 關閉監聽器
	defer ln.Close()

	for {
		// 等待並回傳一個連線
		conn, err := ln.Accept()
		if err != nil {
			log.Println(err)
			continue
		}
		// 處理連線
		go handle(conn)
	}
}

func handle(conn net.Conn) {
	// 寫入訊息
	fmt.Fprintln(conn, "Hello from TCP server")
	// 讀取訊息
	scanner := bufio.NewScanner(conn)
	for scanner.Scan() {
		log.Println(scanner.Text())
	}
	// 關閉連線
	defer conn.Close()
}
```

使用 `telnet` 指令測試。

```BASH
telnet localhost 8080
```
